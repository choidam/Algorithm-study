```c++
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int m, n, k;
int map[100][100] = {0,};
bool visited[100][100] = {false,};

const int dy[] = {-1,1,0,0};
const int dx[] = {0,0,-1,1};

vector<int> ans;

int dfs(int y, int x){
    int cnt = 1;
    
    for(int i=0; i<4; i++){
        int ny = y + dy[i];
        int nx = x + dx[i];
        
        if(ny<0 || nx<0 || ny>=m || nx>=n) continue;
        if(visited[ny][nx]) continue;
        if(!visited[ny][nx] && map[ny][nx]==0){
            visited[ny][nx] = true;
            cnt += dfs(ny,nx);
        }
    }
    
    return cnt;
}

int main(void){
    cin >> m >> n >> k;
    int x1, y1, x2, y2;
    for(int i=0; i<k; i++){
        cin >> x1 >> y1 >> x2 >> y2;
        
        for(int x=x1; x<x2; x++){
            for(int y=y1; y<y2; y++){
                map[y][x] = 1;
            }
        }
    }
    
    for(int i=0; i<m; i++){
        for(int j=0; j<n; j++){
            if(!visited[i][j] && map[i][j]==0){
                visited[i][j] = true;
                ans.push_back(dfs(i,j));
            }
        }
    }

    cout << ans.size() << endl;
    
    sort(ans.begin(), ans.end());
    for(int i=0; i<ans.size(); i++){
        cout << ans[i] << " ";
    }
    
    return 0;
}
```
